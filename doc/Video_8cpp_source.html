28 class Video::Impl {
29 public:
30  Impl(const std::string& filename)
31  : decoder(THEORAPLAY_startDecodeFile(filename.c_str(), 60, THEORAPLAY_VIDFMT_IYUV)),
32  startTime(jngl::getTime()) {
33  if (!decoder) {
34  throw std::runtime_error("Failed to start decoding " + filename + "!");
35  }
36 
37  while (!video) {
38  video = THEORAPLAY_getVideo(decoder);
39  }
40  timePerFrame = 1. / double(video->fps);
41  assert(timePerFrame > 0);
42 
43  GetAudio();
44  alGenSources(1, &source);
45  checkAlError();
46  alListener3f(AL_POSITION, 0.0f, 0.0f, 0.0f);
47  checkAlError();
48  alSource3f(source, AL_POSITION, 0.0f, 0.0f, 0.0f);
49  checkAlError();
50 
51  while (!audio) {
52  audio = THEORAPLAY_getAudio(decoder);
53  }
54 
55  switch (audio->channels) {
56  case 1:
57  format = AL_FORMAT_MONO16;
58  break;
59  case 2:
60  format = AL_FORMAT_STEREO16;
61  break;
62  default:
63  throw std::runtime_error("Unsupported number of channels.");
64  }
65  }
66 
67  void draw() {
68  double now = jngl::getTime() - startTime;
69  if (!video) {
70  video = THEORAPLAY_getVideo(decoder);
71  }
72  if (video and double(video->playms) / 1000. <= now) {
73  if (now - double(video->playms) / 1000. >= timePerFrame) {
74  // Skip frames to catch up, but keep track of the last one in case we catch up to a
75  // series of dupe frames, which means we'd have to draw that final frame and then
76  // wait for more.
77  const THEORAPLAY_VideoFrame* last = video;
78  while ((video = THEORAPLAY_getVideo(decoder)) != nullptr) {
79  jngl::debug("\x1b[33mskipped frame at ");
80  jngl::debug(last->playms);
81  jngl::debugLn("ms\x1b[0m");
82  THEORAPLAY_freeVideo(last);
83  last = video;
84  if (now - double(video->playms) / 1000. < timePerFrame) { break; }
85  }
86 
87  if (!video) {
88  video = last;
89  }
90  }
91  if (!shaderProgram) {
92  Shader vertexShader(R"(#version 300 es
93  in mediump vec2 position;
94  in mediump vec2 inTexCoord;
95  uniform mediump mat3 modelview;
96  uniform mediump mat4 projection;
97  out mediump vec2 texCoord;
98 
99  void main() {
100  vec3 tmp = modelview * vec3(position, 1);
101  gl_Position = projection * vec4(tmp.x, tmp.y, 0, 1);
102  texCoord = inTexCoord;
103  })", Shader::Type::VERTEX
104  );
105  Shader fragmentShader(R"(#version 300 es
106  uniform sampler2D texY;
107  uniform sampler2D texU;
108  uniform sampler2D texV;
109 
110  in mediump vec2 texCoord;
111 
112  out lowp vec4 outColor;
113 
114  void main() {
115  lowp float y = texture(texY, texCoord).r;
116  lowp float u = texture(texU, texCoord).r - 0.5;
117  lowp float v = texture(texV, texCoord).r - 0.5;
118  y = 1.1643 * (y - 0.0625);
119  outColor = vec4(
120  y + 1.5958 * v,
121  y - 0.39173 * u - 0.81290 * v,
122  y + 2.017 * u,
123  1.0
124  );
125  })", Shader::Type::FRAGMENT
126  );
127  shaderProgram = std::make_unique<ShaderProgram>(vertexShader, fragmentShader);
128  modelviewUniform = shaderProgram->getUniformLocation("modelview");
129  const auto projectionUniform =
130  shaderProgram->getUniformLocation("projection");
131  const auto tmp = shaderProgram->use();
132  glUniform1i(shaderProgram->getUniformLocation("texU"), 1);
133  glUniform1i(shaderProgram->getUniformLocation("texV"), 2);
134  glUniformMatrix4fv(projectionUniform, 1, GL_TRUE, &opengl::projection.a[0][0]);
135 
136  textureY = opengl::genAndBindTexture();
137  glTexImage2D(GL_TEXTURE_2D, 0, GL_RED, video->width, video->height, 0, GL_RED,
138  GL_UNSIGNED_BYTE, nullptr);
139 
140  textureU = opengl::genAndBindTexture();
141  // U and V components are a quarter of the size of the video:
142  glTexImage2D(GL_TEXTURE_2D, 0, GL_RED, video->width / 2, video->height / 2, 0,
143  GL_RED, GL_UNSIGNED_BYTE, nullptr);
144 
145  textureV = opengl::genAndBindTexture();
146  // U and V components are a quarter of the size of the video:
147  glTexImage2D(GL_TEXTURE_2D, 0, GL_RED, video->width / 2, video->height / 2, 0,
148  GL_RED, GL_UNSIGNED_BYTE, nullptr);
149 
150  const float preciseWidth = video->width; // TODO: scaling
151  const float preciseHeight = video->height;
152  const static GLfloat vertexes[] = {
153  -preciseWidth / 2, -preciseHeight / 2,
154  0, 0, // texture coordinates
155  -preciseWidth / 2, preciseHeight / 2,
156  0, 1, // texture coordinates
157  preciseWidth / 2, preciseHeight / 2,
158  1, 1, // texture coordinates
159  preciseWidth / 2, -preciseHeight / 2,
160  1, 0 // texture coordinates
161  };
162  glGenVertexArrays(1, &vao);
163  glBindVertexArray(vao);
164 
165  glGenBuffers(1, &vertexBuffer);
166  glBindBuffer(GL_ARRAY_BUFFER, vertexBuffer);
167  glBufferData(GL_ARRAY_BUFFER, 16 * sizeof(GLfloat), vertexes, GL_STATIC_DRAW);
168 
169  const GLint posAttrib = shaderProgram->getAttribLocation("position");
170  glVertexAttribPointer(posAttrib, 2, GL_FLOAT, GL_FALSE, 4 * sizeof(float), nullptr);
171  glEnableVertexAttribArray(posAttrib);
172 
173  const GLint texCoordAttrib = shaderProgram->getAttribLocation("inTexCoord");
174  glVertexAttribPointer(texCoordAttrib, 2, GL_FLOAT, GL_FALSE, 4 * sizeof(float),
175  reinterpret_cast<void*>(2 * sizeof(float)));
176  glEnableVertexAttribArray(texCoordAttrib);
177  }
178 
179  glBindTexture(GL_TEXTURE_2D, textureY);
180  glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, video->width, video->height, GL_RED,
181  GL_UNSIGNED_BYTE, video->pixels);
182 
183  assert(video->width % 2 == 0 & video->height % 2 == 0);
184  glBindTexture(GL_TEXTURE_2D, textureU);
185  glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, video->width / 2, video->height / 2, GL_RED,
186  GL_UNSIGNED_BYTE, video->pixels + video->width * video->height);
187 
188  glBindTexture(GL_TEXTURE_2D, textureV);
189  glTexSubImage2D(GL_TEXTURE_2D, 0, 0, 0, video->width / 2, video->height / 2, GL_RED,
190  GL_UNSIGNED_BYTE,
191  video->pixels + std::lround(1.25 * video->width * video->height));
192 
193  THEORAPLAY_freeVideo(video);
194  video = nullptr;
195  }
196  if (!audio) {
197  audio = THEORAPLAY_getAudio(decoder);
198  }
199  while (audio) {
200  ALint processed;
201  alGetSourcei(source, AL_BUFFERS_PROCESSED, &processed);
202  checkAlError();
203 
204  assert(processed >= 0);
205 
206  if (processed > 0) {
207  // Reuse a processed buffer
208  alSourceUnqueueBuffers(source, 1, &buffers.front());
209  checkAlError();
210 
211  queueAudio(buffers.front());
212  buffers.pop_front();
213 
214  ALint state;
215  alGetSourcei(source, AL_SOURCE_STATE, &state);
216  if (state != AL_PLAYING) {
217 
218  while (true) {
219  alGetSourcei(source, AL_BUFFERS_PROCESSED, &processed);
220  checkAlError();
221  if (processed == 0) {
222  break;
223  }
224  alSourceUnqueueBuffers(source, 1, &buffers.front());
225  checkAlError();
226  alDeleteBuffers(1, &buffers.front());
227  buffers.pop_front();
228  }
229 
230  alSourcePlay(source);
231  checkAlError();
232  jngl::debugLn("WARNING: Audio buffer underrun!");
233  }
234  } else {
235  ALuint buffer;
236  alGenBuffers(1, &buffer);
237  checkAlError();
238  queueAudio(buffer);
239  if (buffers.size() == 1) {
240  alSourcePlay(source);
241  checkAlError();
242  }
243  }
244  }
245  if (shaderProgram) {
246  auto _ = shaderProgram->use();
247  glUniformMatrix3fv(modelviewUniform, 1, GL_TRUE, &opengl::modelview.a[0][0]);
248  glBindVertexArray(vao);
249 
250  glActiveTexture(GL_TEXTURE0);
251  glBindTexture(GL_TEXTURE_2D, textureY);
252 
253  glActiveTexture(GL_TEXTURE0 + 1);
254  glBindTexture(GL_TEXTURE_2D, textureU);
255 
256  glActiveTexture(GL_TEXTURE0 + 2);
257  glBindTexture(GL_TEXTURE_2D, textureV);
258 
259  glDrawArrays(GL_TRIANGLE_FAN, 0, 4);
260  }
261  }
262 
263  ~Impl() {
264  alSourceStop(source);
265  checkAlError();
266  for (auto buffer : buffers) {
267  alSourceUnqueueBuffers(source, 1, &buffer);
268  checkAlError();
269  alDeleteBuffers(1, &buffer);
270  checkAlError();
271  }
272  alDeleteSources(1, &source);
273  checkAlError();
274  }

279 private:
280  void queueAudio(ALuint buffer) {
281  auto pcm = std::make_unique<int16_t[]>(audio->frames * audio->channels);
282  for (size_t i = 0; i < audio->frames * audio->channels; ++i) {
283  const float sample = std::clamp(audio->samples[i], -1.f, 1.f);
284  pcm[i] = sample * 32767.f;
285  }
286 
287  alBufferData(buffer, format, pcm.get(), audio->frames * audio->channels * sizeof(int16_t),
288  audio->freq);
289  checkAlError();
290  alSourceQueueBuffers(source, 1, &buffer);
291  checkAlError();
292 
293  buffers.push_back(buffer);
294 
295  THEORAPLAY_freeAudio(audio);
296  audio = THEORAPLAY_getAudio(decoder);
297  }
