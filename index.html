Just a neat game library

An easy to use C++ game library for iOS, Android, Windows, Mac and Linux!

Example

#include <jngl.hpp>

class HelloWorld : public jngl::Work {
public:
	void step() {}
	void draw() const {
		text.draw();
	}
private:
	jngl::Text text("Hello World!");
};

int main() {
	jngl::setWork(new HelloWorld);
	jngl::mainLoop();
}
