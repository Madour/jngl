Example

#include <jngl.hpp>

class HelloWorld : public jngl::Work {
public:
    void step() {}
    void draw() const {
        text.draw();
    }
private:
    jngl::Text text("Hello World!");
};

int main() {
    jngl::setWork(new HelloWorld);
    jngl::mainLoop();
}
